import React, { useState } from 'react';
import './LoginSignup.css';
import google from '../Assets/google.png';
import fb from '../Assets/fb.png';
import github from '../Assets/github.png';
import insta from '../Assets/insta.png';
import right from '../Assets/right_logo.png';
import arrow_back from '../Assets/arrow_back.png';

const LoginSignup = () => {
  const [action, setAction] = useState("Sign Up");

  
  const headings = {
    "Sign Up": "Create Account",
    "Login": "Welcome Back",
    "Forgot_Password": "",
    "reset_btn": "Reset Password",
    "verify_btn": "Verify Email",
    "confirm": "Confirm Password",
    "update_btn": "Password Updated"
  };

  return (
    <div className='container'>
      <div className='box'>

        <h1>{headings[action] || ""}</h1>

        <div className='text-box'>
          {action==="Sign Up" && (
              <input type="text" placeholder='Name' />
          )}


          {action !== "Forgot_Password" && action!=="reset_btn" && action!=="verify_btn" && action!=="confirm" && action!=="update_btn" && (
            <>
              <input type="email" placeholder="Email Id" />
              <input type="password" placeholder="Password" />
              <span className='fgt'>  
                Forgot password? 
                <a onClick={()=>setAction("Forgot_Password")} href="#">Click here</a>
              </span>
            </>
          )}




          {action==="Forgot_Password" && (
            <>
              <div className='arrow'>
                <img src={arrow_back} alt="" />
              </div>
              <div className='forgo'>
                <h2>Forgot password</h2>
                <p>Please enter your email to reset the password</p>
                <input type="email" placeholder='Enter your email' />
                <button className={action} onClick={() => setAction("reset_btn")}>Reset Password</button>
              </div>
            </>
          )}



          {action==="reset_btn" && (
            <div className='reset'>
              <h4>Check your email</h4>
              <p>We sent a reset link. Enter the 5-digit code mentioned in the email</p>
              <div>
                <input type="number" />
                <input type="number" />
                <input type="number" />
                <input type="number" />
                <input type="number" />
              </div>
              <button className={action} onClick={() => setAction("verify_btn")}>Verify Code</button>
              <span>Haven't got the email yet? <a href="#">Resend email</a></span>
            </div>
          )}




          {action==="verify_btn" && (
            <div className='confirm'>
              <h4>Password reset</h4>
              <p>Your password has been successfully reset. Click confirm to set a new password.</p>
              <button className={action} onClick={() => setAction("confirm")}>Confirm</button>
            </div>
          )}



          {action==="confirm" && (
            <div className='newpwsd'>
              <div>
                <h2>Set a new Password</h2>
                <p>Create a new password. Ensure it differs from previous ones for security.</p>
              </div>
              <div className='pc'>
                <h4>Password</h4>
                <input type="password" placeholder='Enter your new password'/>
                <h4>Confirm Password</h4>
                <input type="password" placeholder='Re-enter password'/>
              </div>
              <div className='up_btn'>
                <button className={action} onClick={() => setAction("update_btn")}>Update Password</button>
              </div>
            </div>
          )}



          {action==="update_btn" && (
            <div className='last'>
              <img src={right} alt="" />
              <h2>Successful</h2>
              <p>Congratulations! Your password has been changed. Click continue to login.</p>
              <div className='last_btn'>
                <button className={action} onClick={() => setAction("Sign Up")}>Continue</button>
              </div>
            </div>
          )}



          {action!=="verify_btn" && action!=="confirm" && action!=="Forgot_Password" && action!=="reset_btn" && action!=="update_btn" && (
            <>
              <div className='icons-box'>
                <div>
                  <a href="https://www.google.com" target="_blank" rel="noopener noreferrer">
                    <img className="pic1" src={google} alt="Google" />
                  </a>
                </div>
                <div>
                  <a href="https://www.facebook.com" target="_blank" rel="noopener noreferrer">
                    <img className="pic2" src={fb} alt="Facebook" />
                  </a>
                </div>
                <div>
                  <a href="https://www.github.com" target="_blank" rel="noopener noreferrer">
                    <img className="pic3" src={github} alt="GitHub" />
                  </a>
                </div>
                <div>
                  <a href="https://www.instagram.com" target="_blank" rel="noopener noreferrer">
                    <img className="pic4" src={insta} alt="Instagram" />
                  </a>
                </div>
              </div>

              <div className='btn'>
                <button id="btn1" className={action==="Sign Up" ? "gray":"submit"} onClick={()=>setAction("Sign Up")}>Sign Up</button>
                <button id="btn2" className={action==="Login" ? "gray":"submit"} onClick={()=>setAction("Login")}>Login</button>
              </div>
            </>
          )}
        </div>
      </div>
    </div>
  );
};

export default LoginSignup;
